* CentOS 5:

Install GIT:
rpm --import http://apt.sw.be/RPM-GPG-KEY.dag.txt
wget http://pkgs.repoforge.org/rpmforge-release/rpmforge-release-0.5.3-1.el5.rf.i386.rpm
rpm -i rpmforge-release-0.5.3-1.el5.rf.i386.rpm
yum install git

Install Puppet:
rpm -ivh http://yum.puppetlabs.com/el/5/products/i386/puppetlabs-release-5-7.noarch.rpm
yum install puppet

Install PostgreSQL repository
wget http://yum.postgresql.org/9.2/redhat/rhel-5-i386/pgdg-centos92-9.2-6.noarch.rpm
rpm -i pgdg-centos92-9.2-6.noarch.rpm

Install nginx repository:
wget http://nginx.org/packages/centos/5/noarch/RPMS/nginx-release-centos-5-0.el5.ngx.noarch.rpm
rpm -ivh nginx-release-centos-5-0.el5.ngx.noarch.rpm

Install required Puppet modules:
puppet module install alup/rbenv
puppet module install thias/nginx
